from os import chdir
from glob import glob

def dfs(s):
	s = s[0]
	#print('s = ', s)
	typ = s[-1]
	#print(typ)
	if typ == '+':
		return dfs(glob(s+"/0_*")) + dfs(glob(s+"/1_*"))
	if typ == 'x':
		return dfs(glob(s+"/0_*")) * dfs(glob(s+"/1_*"))
	if typ == 'r':
		#print('path = ', s)
		return int(open(s).read())
for i in range(37):
	chdir("flag["+str(i)+"]/")
	print(chr(dfs(glob("0_*"))), end = '')
	chdir("..")

#!/usr/bin/env python3
from Crypto.Util.number import *
import os
import sys
import socket
from fractions import Fraction

host = "34.82.101.212"
port = 20001

s = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
s.connect((host, port))
print('connect success')

def ask(C):
	s.send(bytes('2\n','utf-8'))
	_ = s.recv(1024)
	s.send(bytes(f'{C}\n','utf-8'))
	raw = s.recv(1024).split()
	#print(raw)
	return int(raw[2])

def main():
	print('main start')
	_ = s.recv(1024)
	s.send(bytes('1\n','utf-8'))
	raw = s.recv(1024).split()
	C = int(raw[2]); N = int(raw[5])
	#exit()

	mp = {}
	for i in range(3):
		mp[-N * i % 3] = i
	ciph3 = pow(3, 65537, N)

	L = Fraction(0, 1)
	R = Fraction(N, 1)
	while R-L > 0.01:
		C = C * ciph3 % N
		K = mp[ask(C)]
		I = (R-L)/3
		L = L + I*K
		R = L + I
		print(L, R)
	print(round(L))
	print(round(R))
	# L,R is plain text

main()

(Update: 2021/8/15)

#include <bits/stdc++.h>

using ll = long long;

signed main() {
    ll n = 5298309920314;
    std::cout << abs(n) << std::endl; // overflow!
    std::cout << llabs(n) << std::endl;
    std::cout << std::abs(n) << std::endl;
}

#include <iostream>
#include <map>

using namespace std;

struct F {
    int a[2];
    F() : a{0, 1} {}
};

void print(int a[2]) {
    for (int i = 0; i < 2; i++)
        cerr << a[i];
    cerr << '\n';
}

map<int,F> mp;
F GET(int i) {
    return mp[i];
}
void safe() {
    cerr << "--- safe BEGIN ---\n";
    auto B = GET(0);
    auto a = B.a;
    print(a);
    cerr << "--- safe END -----\n";
}
void notsafe() {
    cerr << "--- notsafe BEGIN ---\n";
    auto a = GET(0).a;
    print(a);
    cerr << "--- notsafe END -----\n";
}
signed main() {
    F A;
    mp[0] = A;
    print(A.a);
    safe();
    notsafe();
}
